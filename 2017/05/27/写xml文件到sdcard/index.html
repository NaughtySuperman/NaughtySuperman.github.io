title: Android 文件xml写入sdcard
date: 2017-6-20 10:20:15

tags: Android

/**
 * 把数据以xml的格式写到sdcard里
 */
private void saveDataToScardXml(){try {
        //这个目录下的文件是可以被手动在应用程序信息里面的是可以清除掉  /data/data/应用程序包名/cahce/aaa.txt
        File cacheDir = getCacheDir();
        File file = new File(cacheDir, "aaa.txt");
        FileInputStream fis = new FileInputStream(file);

        BufferedReader br = new BufferedReader(new InputStreamReader(fis));
        String str = br.readLine();
        fis.close();
        br.close();
        System.out.println("cache ： "+str);
    } catch (Exception e) {
        e.printStackTrace();
    }

    try {
        //step1:得到一个序列化器 即 一支笔
        XmlSerializer serializer = Xml.newSerializer();
        //step2: 写
        File file = new File(Environment.getExternalStorageDirectory(), "aa.xml");
        OutputStream os = new FileOutputStream(file);
        serializer.setOutput(os, "utf-8");
        //开始创建xml
        serializer.startDocument("utf-8", true);

        serializer.startTag("", "Users");

        for (int i = 0; i < 5; i++) {

            serializer.startTag("", "user");

                serializer.startTag("", "name");
                serializer.text("流浪");
                serializer.endTag("", "name");

                serializer.startTag("", "sex");
                serializer.text("女");
                serializer.endTag("", "sex");

                serializer.startTag("", "height");
                serializer.text("160cm");
                serializer.endTag("", "height");

            serializer.endTag("", "user");
        }

        serializer.endTag("", "Users");
    //文档写入完毕
    serializer.endDocument();            
    } catch (Exception e) {
        e.printStackTrace();
    }

}
