Dom解析
private List<Book> domResolverXml(){
    try {
        //step1: 创建一个dom解析器工厂
        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
        //step2: 利用工厂对象创建一个解析器对象
        DocumentBuilder db = factory.newDocumentBuilder();
        //step3：开始xml文件
        InputStream is = getAssets().open("Books.xml");
        //Document 文档
        Document document = db.parse(is);
        //获得所有为"Book"标签的元素
        NodeList noList = document.getElementsByTagName("Book");//4
        //创建一个存放书本信息的集合
        List<Book> books = new ArrayList<Book>();

        for (int i = 0; i < noList.getLength(); i++) {
            //变成元素
            Book book = new Book();
            Element element = (Element) noList.item(i);
            String name = element.getElementsByTagName("name").item(0).getTextContent();//红楼梦
            String price = element.getElementsByTagName("price").item(0).getTextContent();
            String author = element.getElementsByTagName("author").item(0).getTextContent();
            //赋值
            book.setName(name);
            book.setPrice(price);
            book.setAuthor(author);
            //添加到集合
            books.add(book);
        }
        return books;
    } catch (Exception e) {
        e.printStackTrace();
    }
    return null;


}
