#数据库事务
SQLite事务 特点：同时执行，或者同时不执行
db.beginTransaction();
try {
db.execSQL(“update person set phone = 83 where id = 2”);
//try里面的代码同时成功，或者不成功：比如取钱，交易的时候的操作
db.execSQL(“update person set phone = 143 where id = 3”);
db.setTransactionSuccessful();//设置事务成功，必须要设置，必须要在endTransaction()调用
} finally {
db.endTransaction();
}

#数据库升级
简单的方式就是添加一列，一般不推荐这样做
String SQL=“alter table 表名 add 列名 varchar（20）”
/**

 * 思路  推介做法
 * 1.将perosn表改成临时表
 * alter table person rename to temp_person
 * 2.创建一个新表
 * db.execSQL("create table person (id Integer primary key autoincrement,name varchar(20),phone varchar(20),sex varchar(5))");
 * 3.检索旧表的数据，插入到新表
 * 查询就表
 * 4.删除临时表
 *  删除旧表
 * 
 */

//1.
db.execSQL("alter table person rename to temp_person");
//2.
db.execSQL("create table person (id Integer primary key autoincrement,name varchar(20),phone varchar(20),sex varchar(5))");
//3.
Cursor cursor = db.query("temp_person", null, null, null, null, null, null);

while(cursor.moveToNext()){

    String name = cursor.getString(cursor.getColumnIndex("name"));
    String phone = cursor.getString(cursor.getColumnIndex("phone"));

    //往新表中插入
    ContentValues values = new ContentValues();
    values.put("name", name);
    values.put("phone", phone);
    db.insert("person", null, values);
}
//4.删除旧表
db.execSQL("drop table temp_person");
